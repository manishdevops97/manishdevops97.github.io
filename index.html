import json import boto3 import time from datetime import datetime ses = boto3.client('ses', region_name='ap-south-1') dynamodb = boto3.resource('dynamodb', region_name='ap-south-1') table = dynamodb.Table('ContactQueries') SENDER = "cloud.manishjangra97@gmail.com" RECEIVER = "aws-email@boxfi.uk" def lambda_handler(event, context): try: # 1. Parse request body safely body = event.get('body', '{}') # Handle case when body is already dict if isinstance(body, str): body = json.loads(body) subject = body.get('subject', '') message = body.get('message', '') if not subject or not message: return response(400, {"error": "Subject and Message required"}) # 2. Send Email via SES ses.send_email( Source=SENDER, Destination={ 'ToAddresses': [RECEIVER] }, Message={ 'Subject': {'Data': subject}, 'Body': {'Text': {'Data': message}} } ) # 3. Store in DynamoDB table.put_item( Item={ 'id': str(int(time.time() * 1000)), 'subject': subject, 'message': message, 'createdAt': datetime.utcnow().isoformat() } ) return response(200, {"message": "Email sent & stored successfully"}) except Exception as e: print("ERROR:", str(e)) # CloudWatch logs return response(500, {"error": str(e)}) # 🔥 Common response formatter (VERY IMPORTANT) def response(status_code, body): return { "statusCode": status_code, "headers": { "Content-Type": "application/json", "Access-Control-Allow-Origin": "*", # CORS fix "Access-Control-Allow-Headers": "*", "Access-Control-Allow-Methods": "POST,OPTIONS" }, "body": json.dumps(body) }
